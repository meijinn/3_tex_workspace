\mysection{演習2}
\subsection{実行プログラム}
実行プログラムをソースコード\ref{s2}に示す。
\begin{lstlisting}[caption=演習2のプログラム,label=s2]
#include <stdio.h>
#include <process.h>

int main()
{
	int i;
	FILE *fp;

	if((fp = fopen("test.dat", "wt")) == NULL){
		printf("Cannot Open test.dat!!\n");
		exit(1);
	}
	
	for(i = 0; i < 100; i++){
		fprintf(fp, "%3d\n", i);
	}
	fclose(fp);
	return 0;
}  
\end{lstlisting}
